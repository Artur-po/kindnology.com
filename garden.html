        <script>
        (async function() {
        // DOM elements
        const operatorsEl = document.getElementById('operators');
        const experienceEl = document.getElementById('experience');
        const insightsEl = document.getElementById('insights');
        const bonusEl = document.getElementById('bonus');
        const addressInput = document.getElementById('address');
        const experienceInput = document.getElementById('experienceInput');
        const addressStatus = document.getElementById('addressStatus');
        const startBtn = document.getElementById('startBtn');
        // Global data store
        let passportData = {}; // address -> totalProfit
        let totalOperators = 0;
        let totalExperience = 0;
        let usedBonus = 0; // will be read from localStorage or contract
        // Load passports.csv
        async function loadPassport() {
        try {
        const response = await fetch('passports.csv');
        const csvText = await response.text();
        const lines = csvText.split('\n').filter(line => line.trim() !== '');
        const headers = lines[0].split(',');
        // expected headers: address,total_profit,...
        for (let i = 1; i < lines.length; i++) {
        const cols = lines[i].split(',');
        if (cols.length < 2) continue;
        const addr = cols[0].trim().toLowerCase();
        const profit = parseFloat(cols[1]);
        if (addr && !isNaN(profit)) {
        passportData[addr] = profit;
        }
        }
        // calculate aggregates
        totalOperators = Object.keys(passportData).length;
        totalExperience = Object.values(passportData).reduce((a, b) => a + b, 0);
        operatorsEl.textContent = totalOperators.toLocaleString();
        experienceEl.textContent = Math.round(totalExperience).toLocaleString();
        // insights: we don't have direct column, show a placeholder (can be updated later)
        insightsEl.textContent = '624'; // you can change this or load from another file
        } catch (e) {
        console.error('Failed to load passports.csv', e);
        operatorsEl.textContent = '?';
        experienceEl.textContent = '?';
        }
        }
        // Load used bonus slots from localStorage (for demo)
        function loadBonus() {
        const saved = localStorage.getItem('usedBonusSlots');
        usedBonus = saved ? parseInt(saved) : 0;
        const left = Math.max(0, 100 - usedBonus);
        bonusEl.textContent = left;
        }
        // Save used bonus
        function useBonusSlot() {
        if (usedBonus < 100) {
        usedBonus++;
        localStorage.setItem('usedBonusSlots', usedBonus);
        bonusEl.textContent = (100 - usedBonus);
        return true;
        }
        return false;
        }
        // Check address in passport
        function checkAddress() {
        const rawAddr = addressInput.value.trim();
        if (!rawAddr.startsWith('0x') || rawAddr.length !== 42) {
        addressStatus.textContent = '❌ Invalid address format.';
        addressStatus.className = 'status-message status-error';
        experienceInput.readOnly = false;
        return;
        }
        const addrLower = rawAddr.toLowerCase();
        const foundProfit = passportData[addrLower];
        if (foundProfit !== undefined) {
        addressStatus.textContent = `✅ Welcome back. Your recorded experience: ${foundProfit.toFixed(2)} ETH.`;
        addressStatus.className = 'status-message status-success';
        experienceInput.value = foundProfit.toFixed(2);
        experienceInput.readOnly = true;
        } else {
        addressStatus.textContent = '❌ Address not found in our records. Please enter your experience manually.';
        addressStatus.className = 'status-message status-error';
        experienceInput.readOnly = false;
        experienceInput.value = '';
        }
        }
        // Handle start journey
        function startJourney() {
        if (!window.ethereum) {
        alert('Please install MetaMask to continue.');
        return;
        }
        const addr = addressInput.value.trim();
        if (!addr.startsWith('0x') || addr.length !== 42) {
        alert('Enter a valid Ethereum address.');
        return;
        }
        const experience = parseFloat(experienceInput.value);
        if (isNaN(experience) || experience <= 0) {
        alert('Enter a positive experience amount.');
        return;
        }
        // check bonus availability
        if (usedBonus >= 100) {
        alert('Sorry, all bonus slots are taken. You can still join without the extra boost.');
        } else {
        // simulate reservation (in production call contract)
        if (confirm('This will reserve a bonus slot. Continue?')) {
        useBonusSlot();
        alert('Congratulations! You’ve reserved a bonus slot. Soon you’ll be able to finalize the transaction.');
        }
        }
        }
        // Event listeners
        addressInput.addEventListener('blur', checkAddress);
        startBtn.addEventListener('click', startJourney);
        // Initial load
        await loadPassport();
        loadBonus();
        // Pre-fill example if needed (optional)
        // addressInput.value = '0xae2fc483527b8ef99eb5d9b44875f005ba1fae13';
        // checkAddress();
        })();
